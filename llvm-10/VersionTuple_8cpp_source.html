38 static bool parseInt(StringRef &input, unsigned &value) {
39  assert(value == 0);
40  if (input.empty())
41  return true;
42 
43  char next = input[0];
44  input = input.substr(1);
45  if (next < '0' || next > '9')
46  return true;
47  value = (unsigned)(next - '0');
48 
49  while (!input.empty()) {
50  next = input[0];
51  if (next < '0' || next > '9')
52  return false;
53  input = input.substr(1);
54  value = value * 10 + (unsigned)(next - '0');
55  }
56 
57  return false;
58 }

62
