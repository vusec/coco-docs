16 /// This is a llvm local version of __cxa_demangle. Other than the name and
17 /// being in the llvm namespace it is identical.

53 /// "Partial" demangler. This supports demangling a string into an AST
54 /// (typically an intermediate stage in itaniumDemangle) and querying certain
55 /// properties or partially printing the demangled name.

62  /// Demangle into an AST. Subsequent calls to the rest of the member functions
63  /// implicitly operate on the AST this produces.
64  /// \return true on error, false otherwise
65  bool partialDemangle(const char *MangledName);

67  /// Just print the entire mangled name into Buf. Buf and N behave like the
68  /// second and third parameters to itaniumDemangle.
69  char *finishDemangle(char *Buf, size_t *N) const;

71  /// Get the base name of a function. This doesn't include trailing template
72  /// arguments, ie for "a::b<int>" this function returns "b".

75  /// Get the context name for a function. For "a::b::c", this function returns

78

84  char *getFunctionReturnType(char *Buf, size_t *N) const;
85
