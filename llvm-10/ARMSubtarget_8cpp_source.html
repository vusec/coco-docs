161 void ARMSubtarget::initSubtargetFeatures(StringRef CPU, StringRef FS) {
162  if (CPUString.empty()) {
163  CPUString = "generic";
164 
165  if (isTargetDarwin()) {
166  StringRef ArchName = TargetTriple.getArchName();
167  ARM::ArchKind AK = ARM::parseArch(ArchName);
168  if (AK == ARM::ArchKind::ARMV7S)
169  // Default to the Swift CPU when targeting armv7s/thumbv7s.
170  CPUString = "swift";
171  else if (AK == ARM::ArchKind::ARMV7K)
172  // Default to the Cortex-a7 CPU when targeting armv7k/thumbv7k.
173  // ARMv7k does not use SjLj exception handling.
174  CPUString = "cortex-a7";
175  }
176  }
177 
178  // Insert the architecture feature derived from the target triple into the
179  // feature string. This is important for setting features that are implied
180  // based on the architecture version.
181  std::string ArchFS = ARM_MC::ParseARMTriple(TargetTriple, CPUString);
182  if (!FS.empty()) {
183  if (!ArchFS.empty())
184  ArchFS = (Twine(ArchFS) + "," + FS).str();
185  else
186  ArchFS = FS;
187  }
188  ParseSubtargetFeatures(CPUString, ArchFS);
189 
190  // FIXME: This used enable V6T2 support implicitly for Thumb2 mode.
191  // Assert this for now to make the change obvious.
192  assert(hasV6T2Ops() || !hasThumb2());
193 
194  // Execute only support requires movt support
195  if (genExecuteOnly()) {
196  NoMovt = false;
197  assert(hasV8MBaselineOps() && "Cannot generate execute-only code for this target");
198  }
199 
200  // Keep a pointer to static instruction cost data for the specified CPU.
201  SchedModel = getSchedModelForCPU(CPUString);
202 
203  // Initialize scheduling itinerary for the specified CPU.
204  InstrItins = getInstrItineraryForCPU(CPUString);
205 
206  // FIXME: this is invalid for WindowsCE
207  if (isTargetWindows())
208  NoARM = true;
209 
210  if (isAAPCS_ABI())
211  stackAlignment = Align(8);
212  if (isTargetNaCl() || isAAPCS16_ABI())
213  stackAlignment = Align(16);
214 
215  // FIXME: Completely disable sibcall for Thumb1 since ThumbRegisterInfo::
216  // emitEpilogue is not ready for them. Thumb tail calls also use t2B, as
217  // the Thumb1 16-bit unconditional branch doesn't have sufficient relocation
218  // support in the assembler and linker to be used. This would need to be
219  // fixed to fully support tail calls in Thumb1.
220  //
221  // For ARMv8-M, we /do/ implement tail calls. Doing this is tricky for v8-M
222  // baseline, since the LDM/POP instruction on Thumb doesn't take LR. This
223  // means if we need to reload LR, it takes extra instructions, which outweighs
224  // the value of the tail call; but here we don't know yet whether LR is going
225  // to be used. We take the optimistic approach of generating the tail call and
226  // perhaps taking a hit if we need to restore the LR.
227 
228  // Thumb1 PIC calls to external symbols use BX, so they can be tail calls,
229  // but we need to make sure there are enough registers; the only valid
230  // registers are the 4 used for parameters. We don't currently do this
231  // case.
232 
233  SupportsTailCall = !isThumb() || hasV8MBaselineOps();
234 
235  if (isTargetMachO() && isTargetIOS() && getTargetTriple().isOSVersionLT(5, 0))
236  SupportsTailCall = false;
237 
238  switch (IT) {
239  case DefaultIT:
240  RestrictIT = hasV8Ops();
241  break;
242  case RestrictedIT:
243  RestrictIT = true;
244  break;
245  case NoRestrictedIT:
246  RestrictIT = false;
247  break;
248  }
249 
250  // NEON f32 ops are non-IEEE 754 compliant. Darwin is ok with it by default.
251  const FeatureBitset &Bits = getFeatureBits();
252  if ((Bits[ARM::ProcA5] || Bits[ARM::ProcA8]) && // Where this matters
253  (Options.UnsafeFPMath || isTargetDarwin()))
254  UseNEONForSinglePrecisionFP = true;
255 
256  if (isRWPI())
257  ReserveR9 = true;
258 
259  // If MVEVectorCostFactor is still 0 (has not been set to anything else), default it to 2
260  if (MVEVectorCostFactor == 0)
261  MVEVectorCostFactor = 2;
262 
263  // FIXME: Teach TableGen to deal with these instead of doing it manually here.
264  switch (ARMProcFamily) {
265  case Others:
266  case CortexA5:
267  break;
268  case CortexA7:
269  LdStMultipleTiming = DoubleIssue;
270  break;
271  case CortexA8:
272  LdStMultipleTiming = DoubleIssue;
273  break;
274  case CortexA9:
275  LdStMultipleTiming = DoubleIssueCheckUnalignedAccess;
276  PreISelOperandLatencyAdjustment = 1;
277  break;
278  case CortexA12:
279  break;
280  case CortexA15:
281  MaxInterleaveFactor = 2;
282  PreISelOperandLatencyAdjustment = 1;
283  PartialUpdateClearance = 12;
284  break;
285  case CortexA17:
286  case CortexA32:
287  case CortexA35:
288  case CortexA53:
289  case CortexA55:
290  case CortexA57:
291  case CortexA72:
292  case CortexA73:
293  case CortexA75:
294  case CortexA76:
295  case CortexR4:
296  case CortexR4F:
297  case CortexR5:
298  case CortexR7:
299  case CortexM3:
300  case CortexR52:
301  break;
302  case Exynos:
303  LdStMultipleTiming = SingleIssuePlusExtras;
304  MaxInterleaveFactor = 4;
305  if (!isThumb())
306  PrefLoopLogAlignment = 3;
307  break;
308  case Kryo:
309  break;
310  case Krait:
311  PreISelOperandLatencyAdjustment = 1;
312  break;
313  case NeoverseN1:
314  break;
315  case Swift:
316  MaxInterleaveFactor = 2;
317  LdStMultipleTiming = SingleIssuePlusExtras;
318  PreISelOperandLatencyAdjustment = 1;
319  PartialUpdateClearance = 12;
320  break;
321  }
322 }

349 bool ARMSubtarget::isGVIndirectSymbol(const GlobalValue *GV) const {
350  if (!TM.shouldAssumeDSOLocal(*GV->getParent(), GV))
351  return true;
352

354  // the section that is being relocated. This means we have to use o load even
355  // for GVs that are known to be local to the dso.
356  if (isTargetMachO() && TM.isPositionIndependent() &&
357  (GV->isDeclarationForLinker() || GV->hasCommonLinkage()))
358  return true;
359 
360  return false;

372 bool ARMSubtarget::enableMachineScheduler() const {
373  // The MachineScheduler can increase register usage, so we use more high
374  // registers and end up with more T2 instructions that cannot be converted to
375  // T1 instructions. At least until we do better at converting to thumb1
376  // instructions, on cortex-m at Oz where we are size-paranoid, don't use the
377  // Machine scheduler, relying on the DAG register pressure scheduler instead.
378  if (isMClass() && hasMinSize())
379  return false;
380  // Enable the MachineScheduler before register allocation for subtargets
381  // with the use-misched feature.
382  return useMachineScheduler();
383 }

438 unsigned ARMSubtarget::getGPRAllocationOrder(const MachineFunction &MF) const {
439  // The GPR register class has multiple possible allocation orders, with
440  // tradeoffs preferred by different sub-architectures and optimisation goals.
441  // The allocation orders are:
442  // 0: (the default tablegen order, not used)
443  // 1: r14, r0-r13
444  // 2: r0-r7
445  // 3: r0-r7, r12, lr, r8-r11
446  // Note that the register allocator will change this order so that
447  // callee-saved registers are used later, as they require extra work in the
448  // prologue/epilogue (though we sometimes override that).
449 
450  // For thumb1-only targets, only the low registers are allocatable.
451  if (isThumb1Only())
452  return 2;
453 
454  // Allocate low registers first, so we can select more 16-bit instructions.
455  // We also (in ignoreCSRForAllocationOrder) override the default behaviour
456  // with regards to callee-saved registers, because pushing extra registers is
457  // much cheaper (in terms of code size) than using high registers. After
458  // that, we allocate r12 (doesn't need to be saved), lr (saving it means we
459  // can return with the pop, don't need an extra "bx lr") and then the rest of
460  // the high registers.
461  if (isThumb2() && MF.getFunction().hasMinSize())
462  return 3;
463 
464  // Otherwise, allocate in the default order, using LR first because saving it
465  // allows a shorter epilogue sequence.
466  return 1;
467 }

469 bool ARMSubtarget::ignoreCSRForAllocationOrder(const MachineFunction &MF,
470  unsigned PhysReg) const {
471  // To minimize code size in Thumb2, we prefer the usage of low regs (lower
472  // cost per use) so we can use narrow encoding. By default, caller-saved
473  // registers (e.g. lr, r12) are always allocated first, regardless of
474  // their cost per use. When optForMinSize, we prefer the low regs even if
475  // they are CSR because usually push/pop can be folded into existing ones.
476  return isThumb2() && MF.getFunction().hasMinSize() &&
477  ARM::GPRRegClass.contains(PhysReg);
478 }
