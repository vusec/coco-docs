36 inline void PrintTo(const ErrorHolder &Err, std::ostream *Out) {
37  raw_os_ostream OS(*Out);
38  OS << (Err.Success() ? "succeeded" : "failed");
39  if (!Err.Success()) {
40  const char *Delim = " (";
41  for (const auto &Info : Err.Infos) {
42  OS << Delim;
43  Delim = "; ";
44  Info->log(OS);
45  }
46  OS << ")";
47  }
48 }
