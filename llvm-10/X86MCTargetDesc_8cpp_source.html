322 static MCAsmInfo *createX86MCAsmInfo(const MCRegisterInfo &MRI,
323  const Triple &TheTriple,
324  const MCTargetOptions &Options) {
325  bool is64Bit = TheTriple.getArch() == Triple::x86_64;
326 
327  MCAsmInfo *MAI;
328  if (TheTriple.isOSBinFormatMachO()) {
329  if (is64Bit)
330  MAI = new X86_64MCAsmInfoDarwin(TheTriple);
331  else
332  MAI = new X86MCAsmInfoDarwin(TheTriple);
333  } else if (TheTriple.isOSBinFormatELF()) {
334  // Force the use of an ELF container.
335  MAI = new X86ELFMCAsmInfo(TheTriple);
336  } else if (TheTriple.isWindowsMSVCEnvironment() ||
337  TheTriple.isWindowsCoreCLREnvironment()) {
338  MAI = new X86MCAsmInfoMicrosoft(TheTriple);
339  } else if (TheTriple.isOSCygMing() ||
340  TheTriple.isWindowsItaniumEnvironment()) {
341  MAI = new X86MCAsmInfoGNUCOFF(TheTriple);
342  } else {
343  // The default is ELF.
344  MAI = new X86ELFMCAsmInfo(TheTriple);
345  }
346 
347  // Initialize initial frame state.
348  // Calculate amount of bytes used for return address storing
349  int stackGrowth = is64Bit ? -8 : -4;
350 
351  // Initial state of the frame pointer is esp+stackGrowth.
352  unsigned StackPtr = is64Bit ? X86::RSP : X86::ESP;
353  MCCFIInstruction Inst = MCCFIInstruction::createDefCfa(
354  nullptr, MRI.getDwarfRegNum(StackPtr, true), -stackGrowth);
355  MAI->addInitialFrameState(Inst);
356 
357  // Add return address to move list
358  unsigned InstPtr = is64Bit ? X86::RIP : X86::EIP;
359  MCCFIInstruction Inst2 = MCCFIInstruction::createOffset(
360  nullptr, MRI.getDwarfRegNum(InstPtr, true), stackGrowth);
361  MAI->addInitialFrameState(Inst2);
362 
363  return MAI;
364 }

412 bool X86MCInstrAnalysis::clearsSuperRegisters(const MCRegisterInfo &MRI,
413  const MCInst &Inst,
414  APInt &Mask) const {
415  const MCInstrDesc &Desc = Info->get(Inst.getOpcode());
416  unsigned NumDefs = Desc.getNumDefs();
417  unsigned NumImplicitDefs = Desc.getNumImplicitDefs();
418  assert(Mask.getBitWidth() == NumDefs + NumImplicitDefs &&
419  "Unexpected number of bits in the mask!");
420 
421  bool HasVEX = (Desc.TSFlags & X86II::EncodingMask) == X86II::VEX;
422  bool HasEVEX = (Desc.TSFlags & X86II::EncodingMask) == X86II::EVEX;
423  bool HasXOP = (Desc.TSFlags & X86II::EncodingMask) == X86II::XOP;
424 
425  const MCRegisterClass &GR32RC = MRI.getRegClass(X86::GR32RegClassID);
426  const MCRegisterClass &VR128XRC = MRI.getRegClass(X86::VR128XRegClassID);
427  const MCRegisterClass &VR256XRC = MRI.getRegClass(X86::VR256XRegClassID);
428 
429  auto ClearsSuperReg = [=](unsigned RegID) {
430  // On X86-64, a general purpose integer register is viewed as a 64-bit
431  // register internal to the processor.
432  // An update to the lower 32 bits of a 64 bit integer register is
433  // architecturally defined to zero extend the upper 32 bits.
434  if (GR32RC.contains(RegID))
435  return true;
436 
437  // Early exit if this instruction has no vex/evex/xop prefix.
438  if (!HasEVEX && !HasVEX && !HasXOP)
439  return false;
440 
441  // All VEX and EVEX encoded instructions are defined to zero the high bits
442  // of the destination register up to VLMAX (i.e. the maximum vector register
443  // width pertaining to the instruction).
444  // We assume the same behavior for XOP instructions too.
445  return VR128XRC.contains(RegID) || VR256XRC.contains(RegID);
446  };
447 
448  Mask.clearAllBits();
449  for (unsigned I = 0, E = NumDefs; I < E; ++I) {
450  const MCOperand &Op = Inst.getOperand(I);
451  if (ClearsSuperReg(Op.getReg()))
452  Mask.setBit(I);
453  }
454 
455  for (unsigned I = 0, E = NumImplicitDefs; I < E; ++I) {
456  const MCPhysReg Reg = Desc.getImplicitDefs()[I];
457  if (ClearsSuperReg(Reg))
458  Mask.setBit(NumDefs + I);
459  }
460 
461  return Mask.getBoolValue();
462 }

467  // Do a lightweight parsing of PLT entries.
468  std::vector<std::pair<uint64_t, uint64_t>> Result;
469  for (uint64_t Byte = 0, End = PltContents.size(); Byte + 6 < End; ) {
470  // Recognize a jmp.
471  if (PltContents[Byte] == 0xff && PltContents[Byte + 1] == 0xa3) {
472  // The jmp instruction at the beginning of each PLT entry jumps to the
473  // address of the base of the .got.plt section plus the immediate.
474  uint32_t Imm = support::endian::read32le(PltContents.data() + Byte + 2);
475  Result.push_back(
476  std::make_pair(PltSectionVA + Byte, GotPltSectionVA + Imm));
477  Byte += 6;
478  } else if (PltContents[Byte] == 0xff && PltContents[Byte + 1] == 0x25) {
479  // The jmp instruction at the beginning of each PLT entry jumps to the
480  // immediate.
481  uint32_t Imm = support::endian::read32le(PltContents.data() + Byte + 2);
482  Result.push_back(std::make_pair(PltSectionVA + Byte, Imm));
483  Byte += 6;
484  } else
485  Byte++;
486  }
487  return Result;
488 }

522 Optional<uint64_t> X86MCInstrAnalysis::evaluateMemoryOperandAddress(
523  const MCInst &Inst, uint64_t Addr, uint64_t Size) const {
524  const MCInstrDesc &MCID = Info->get(Inst.getOpcode());
525  int MemOpStart = X86II::getMemoryOperandNo(MCID.TSFlags);
526  if (MemOpStart == -1)
527  return None;
528  MemOpStart += X86II::getOperandBias(MCID);
529 
530  const MCOperand &SegReg = Inst.getOperand(MemOpStart + X86::AddrSegmentReg);
531  const MCOperand &BaseReg = Inst.getOperand(MemOpStart + X86::AddrBaseReg);
532  const MCOperand &IndexReg = Inst.getOperand(MemOpStart + X86::AddrIndexReg);
533  const MCOperand &ScaleAmt = Inst.getOperand(MemOpStart + X86::AddrScaleAmt);
534  const MCOperand &Disp = Inst.getOperand(MemOpStart + X86::AddrDisp);
535  if (SegReg.getReg() != 0 || IndexReg.getReg() != 0 || ScaleAmt.getImm() != 1 ||
536  !Disp.isImm())
537  return None;
538 
539  // RIP-relative addressing.
540  if (BaseReg.getReg() == X86::RIP)
541  return Addr + Size + Disp.getImm();
542 
543  return None;
544 }
