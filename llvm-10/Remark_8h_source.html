114 template <typename T>
115 bool operator<(const Optional<T> &LHS, const Optional<T> &RHS) {
116  // Sorting based on optionals should result in all `None` entries to appear
117  // before the valid entries. For example, remarks with no debug location will
118  // appear first.
119  if (!LHS && !RHS)
120  return false;
121  if (!LHS && RHS)
122  return true;
123  if (LHS && !RHS)
124  return false;
125  return *LHS < *RHS;
126 }
