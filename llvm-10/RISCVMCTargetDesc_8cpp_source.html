53 static MCAsmInfo *createRISCVMCAsmInfo(const MCRegisterInfo &MRI,
54  const Triple &TT,
55  const MCTargetOptions &Options) {
56  MCAsmInfo *MAI = new RISCVMCAsmInfo(TT);
57 
58  Register SP = MRI.getDwarfRegNum(RISCV::X2, true);
59  MCCFIInstruction Inst = MCCFIInstruction::createDefCfa(nullptr, SP, 0);
60  MAI->addInitialFrameState(Inst);
61 
62  return MAI;
63 }
