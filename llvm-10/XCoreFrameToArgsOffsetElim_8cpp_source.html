47 bool XCoreFTAOElim::runOnMachineFunction(MachineFunction &MF) {
48  const XCoreInstrInfo &TII =
49  *static_cast<const XCoreInstrInfo *>(MF.getSubtarget().getInstrInfo());
50  unsigned StackSize = MF.getFrameInfo().getStackSize();
51  for (MachineFunction::iterator MFI = MF.begin(), E = MF.end(); MFI != E;
52  ++MFI) {
53  MachineBasicBlock &MBB = *MFI;
54  for (MachineBasicBlock::iterator MBBI = MBB.begin(), EE = MBB.end();
55  MBBI != EE; ++MBBI) {
56  if (MBBI->getOpcode() == XCore::FRAME_TO_ARGS_OFFSET) {
57  MachineInstr &OldInst = *MBBI;
58  Register Reg = OldInst.getOperand(0).getReg();
59  MBBI = TII.loadImmediate(MBB, MBBI, Reg, StackSize);
60  OldInst.eraseFromParent();
61  }
62  }
63  }
64  return true;
