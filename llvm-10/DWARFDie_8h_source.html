32 /// Utility class that carries the DWARF compile/type unit and the debug info
33 /// entry in an object.
34 ///
35 /// When accessing information from a debug info entry we always need to DWARF
36 /// compile/type unit in order to extract the info correctly as some information
37 /// is relative to the compile/type unit. Prior to this class the DWARFUnit and
38 /// the DWARFDebugInfoEntry was passed around separately and there was the
39 /// possibility for error if the wrong DWARFUnit was used to extract a unit
40 /// relative offset. This class helps to ensure that this doesn't happen and
41 /// also simplifies the attribute extraction calls by not having to specify the
42 /// DWARFUnit for each call.
43 class DWARFDie {
44  DWARFUnit *U = nullptr;
45  const DWARFDebugInfoEntry *Die = nullptr;

47 public:

52  explicit operator bool() const { return isValid(); }
53  const DWARFDebugInfoEntry *getDebugInfoEntry() const { return Die; }

396 class reverse_iterator<llvm::DWARFDie::iterator>
397  : public llvm::iterator_facade_base<
398  reverse_iterator<llvm::DWARFDie::iterator>,
399  bidirectional_iterator_tag, const llvm::DWARFDie> {
400 
401 private:
402  llvm::DWARFDie Die;
403  bool AtEnd;
404 
405 public:
406  reverse_iterator(llvm::DWARFDie::iterator It)
407  : Die(It.Die), AtEnd(!It.Die.getPreviousSibling()) {
408  if (!AtEnd)
409  Die = Die.getPreviousSibling();
410  }
411 
412  llvm::DWARFDie::iterator base() const {
413  return llvm::DWARFDie::iterator(AtEnd ? Die : Die.getSibling());
414  }
415 
416  reverse_iterator<llvm::DWARFDie::iterator> &operator++() {
417  assert(!AtEnd && "Incrementing rend");
418  llvm::DWARFDie D = Die.getPreviousSibling();
419  if (D)
420  Die = D;
421  else
422  AtEnd = true;
423  return *this;
424  }
425 
426  reverse_iterator<llvm::DWARFDie::iterator> &operator--() {
427  if (AtEnd) {
428  AtEnd = false;
429  return *this;
430  }
431  Die = Die.getSibling();
432  assert(!Die.isNULL() && "Decrementing rbegin");
433  return *this;
434  }
435 
436  const llvm::DWARFDie &operator*() const {
437  assert(Die.isValid());
438  return Die;
439  }
440 
441  // FIXME: We should be able to specify the equals operator as a friend, but
442  // that causes the compiler to think the operator overload is ambiguous
443  // with the friend declaration and the actual definition as candidates.
444  bool equals(const reverse_iterator<llvm::DWARFDie::iterator> &RHS) const {
445  return Die == RHS.Die && AtEnd == RHS.AtEnd;
446  }
