113 static void fixELFSymbolsInTLSFixupsImpl(const MCExpr *Expr, MCAssembler &Asm) {
114  switch (Expr->getKind()) {
115  case MCExpr::Target:
116  llvm_unreachable("Can't handle nested target expression");
117  break;
118  case MCExpr::Constant:
119  break;
120 
121  case MCExpr::Binary: {
122  const MCBinaryExpr *BE = cast<MCBinaryExpr>(Expr);
123  fixELFSymbolsInTLSFixupsImpl(BE->getLHS(), Asm);
124  fixELFSymbolsInTLSFixupsImpl(BE->getRHS(), Asm);
125  break;
126  }
127 
128  case MCExpr::SymbolRef: {
129  // We're known to be under a TLS fixup, so any symbol should be
130  // modified. There should be only one.
131  const MCSymbolRefExpr &SymRef = *cast<MCSymbolRefExpr>(Expr);
132  cast<MCSymbolELF>(SymRef.getSymbol()).setType(ELF::STT_TLS);
133  break;
134  }
135 
136  case MCExpr::Unary:
137  fixELFSymbolsInTLSFixupsImpl(cast<MCUnaryExpr>(Expr)->getSubExpr(), Asm);
138  break;
139  }
140 }
