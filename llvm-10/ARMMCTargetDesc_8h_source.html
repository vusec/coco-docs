100 enum OperandType {

102  OPERAND_VPRED_N,

104 inline bool isVpred(OperandType op) {
105  return op == OPERAND_VPRED_R || op == OPERAND_VPRED_N;
106 }
107 inline bool isVpred(uint8_t op) {
