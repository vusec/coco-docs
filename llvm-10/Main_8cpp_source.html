61 /// Create a dependency file for `-d` option.
62 ///
63 /// This functionality is really only for the benefit of the build system.
64 /// It is similar to GCC's `-M*` family of options.
65 static int createDependencyFile(const TGParser &Parser, const char *argv0) {
66  if (OutputFilename == "-")
67  return reportError(argv0, "the option -d must be used together with -o\n");
68 
69  std::error_code EC;
70  ToolOutputFile DepOut(DependFilename, EC, sys::fs::OF_None);
71  if (EC)
72  return reportError(argv0, "error opening " + DependFilename + ":" +
73  EC.message() + "\n");
74  DepOut.os() << OutputFilename << ":";
75  for (const auto &Dep : Parser.getDependencies()) {
76  DepOut.os() << ' ' << Dep;
77  }
78  DepOut.os() << "\n";
79  DepOut.keep();
80  return 0;
81 }

83 int llvm::TableGenMain(char *argv0, TableGenMainFn *MainFn) {
84  RecordKeeper Records;
85 
86  // Parse the input file.
87  ErrorOr<std::unique_ptr<MemoryBuffer>> FileOrErr =
88  MemoryBuffer::getFileOrSTDIN(InputFilename);
89  if (std::error_code EC = FileOrErr.getError())
90  return reportError(argv0, "Could not open input file '" + InputFilename +
91  "': " + EC.message() + "\n");
92 
93  // Tell SrcMgr about this buffer, which is what TGParser will pick up.
94  SrcMgr.AddNewSourceBuffer(std::move(*FileOrErr), SMLoc());
95 
96  // Record the location of the include directory so that the lexer can find
97  // it later.
98  SrcMgr.setIncludeDirs(IncludeDirs);
99 
100  TGParser Parser(SrcMgr, MacroNames, Records);
101 
102  if (Parser.ParseFile())
103  return 1;
104 
105  // Write output to memory.
106  std::string OutString;
107  raw_string_ostream Out(OutString);
108  if (MainFn(Out, Records))
109  return 1;
110 
111  // Always write the depfile, even if the main output hasn't changed.
112  // If it's missing, Ninja considers the output dirty. If this was below
113  // the early exit below and someone deleted the .inc.d file but not the .inc
114  // file, tablegen would never write the depfile.
115  if (!DependFilename.empty()) {
116  if (int Ret = createDependencyFile(Parser, argv0))
117  return Ret;
118  }
119 
120  if (WriteIfChanged) {
121  // Only updates the real output file if there are any differences.
122  // This prevents recompilation of all the files depending on it if there
123  // aren't any.
124  if (auto ExistingOrErr = MemoryBuffer::getFile(OutputFilename))
125  if (std::move(ExistingOrErr.get())->getBuffer() == Out.str())
126  return 0;
127  }
128 
129  std::error_code EC;
130  ToolOutputFile OutFile(OutputFilename, EC, sys::fs::OF_None);
131  if (EC)
132  return reportError(argv0, "error opening " + OutputFilename + ":" +
133  EC.message() + "\n");
134  OutFile.os() << Out.str();
135 
136  if (ErrorsPrinted > 0)
137  return reportError(argv0, Twine(ErrorsPrinted) + " errors.\n");
138 
139  // Declare success.
140  OutFile.keep();
141  return 0;
142 }
