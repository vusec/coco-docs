30 static std::string computeDataLayout(const Triple &T) {
31  // Aurora VE is little endian
32  std::string Ret = "e";
33 
34  // Use ELF mangling
35  Ret += "-m:e";
36 
37  // Alignments for 64 bit integers.
38  Ret += "-i64:64";
39 
40  // VE supports 32 bit and 64 bits integer on registers
41  Ret += "-n32:64";
42 
43  // Stack alignment is 64 bits
44  Ret += "-S64";
45 
46  return Ret;
47 }
