42 struct IntrinsicData {
43 
44  uint16_t Id;
45  IntrinsicType Type;
46  uint16_t Opc0;
47  uint16_t Opc1;
48 
49  bool operator<(const IntrinsicData &RHS) const {
50  return Id < RHS.Id;
51  }
52  bool operator==(const IntrinsicData &RHS) const {
53  return RHS.Id == Id;
54  }
55  friend bool operator<(const IntrinsicData &LHS, unsigned Id) {
56  return LHS.Id < Id;
57  }
