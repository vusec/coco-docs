91 /// This is called when a count is seen in the profile.
92 void ProfileSummaryBuilder::addCount(uint64_t Count) {
93  TotalCount += Count;
94  if (Count > MaxCount)
95  MaxCount = Count;
96  NumCounts++;
97  CountFrequencies[Count]++;
