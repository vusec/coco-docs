87  /// Target Operand Flag enum.
88  enum TOF {
89  //===------------------------------------------------------------------===//
90  // PPC Specific MachineOperand flags.
91  MO_NO_FLAG,
92 
93  /// On a symbol operand "FOO", this indicates that the reference is actually
94  /// to "FOO@plt". This is used for calls and jumps to external functions
95  /// and for PIC calls on 32-bit ELF systems.
96  MO_PLT = 1,
97 
98  /// MO_PIC_FLAG - If this bit is set, the symbol reference is relative to
99  /// the function's picbase, e.g. lo16(symbol-picbase).
100  MO_PIC_FLAG = 2,
101 
102  /// MO_NLP_FLAG - If this bit is set, the symbol reference is actually to
103  /// the non_lazy_ptr for the global, e.g. lo16(symbol$non_lazy_ptr-picbase).
104  MO_NLP_FLAG = 4,
105 
106  /// MO_NLP_HIDDEN_FLAG - If this bit is set, the symbol reference is to a
107  /// symbol with hidden visibility. This causes a different kind of
108  /// non-lazy-pointer to be generated.
109  MO_NLP_HIDDEN_FLAG = 8,
110 
111  /// The next are not flags but distinct values.
112  MO_ACCESS_MASK = 0xf0,
113 
114  /// MO_LO, MO_HA - lo16(symbol) and ha16(symbol)
115  MO_LO = 1 << 4,
116  MO_HA = 2 << 4,
117 
118  MO_TPREL_LO = 4 << 4,
119  MO_TPREL_HA = 3 << 4,
120 
121  /// These values identify relocations on immediates folded
122  /// into memory operations.
123  MO_DTPREL_LO = 5 << 4,
124  MO_TLSLD_LO = 6 << 4,
125  MO_TOC_LO = 7 << 4,
126 
127  // Symbol for VK_PPC_TLS fixup attached to an ADD instruction
128  MO_TLS = 8 << 4
129  };
130  } // end namespace PPCII
