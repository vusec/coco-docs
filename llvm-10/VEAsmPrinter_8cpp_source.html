57 void VEAsmPrinter::EmitInstruction(const MachineInstr *MI) {
58 
59  switch (MI->getOpcode()) {
60  default:
61  break;
62  case TargetOpcode::DBG_VALUE:
63  // FIXME: Debug Value.
64  return;
65  }
66  MachineBasicBlock::const_instr_iterator I = MI->getIterator();
67  MachineBasicBlock::const_instr_iterator E = MI->getParent()->instr_end();
68  do {
69  MCInst TmpInst;
70  LowerVEMachineInstrToMCInst(&*I, TmpInst, *this);
71  EmitToStreamer(*OutStreamer, TmpInst);
72  } while ((++I != E) && I->isInsideBundle()); // Delay slot check.
73 }
